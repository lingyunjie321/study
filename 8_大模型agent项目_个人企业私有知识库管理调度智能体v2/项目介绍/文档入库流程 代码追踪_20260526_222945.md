## 概述

文档入库流程是整个系统的入口——用户上传一份文档后，系统将其自动解析、抽取知识、存入向量库和知识图谱。流程由LangGraph StateGraph编排，4个节点串行执行（parse→extract→store_vectors→store_graph），其中store_vectors和store_graph之间也是串行（虽然逻辑上可并行，但LangGraph此处用串行边连接以保证数据完整性）。

涉及的模块：DocParserAgent、KnowledgeExtractAgent、VectorStoreService、KnowledgeGraphService，由`orchestrator/graph.py`的`_build_ingest_graph()`统一编排。

async def parse(self, file_path: str) -> list[DocumentChunk]:
    doc_type = self._classify(file_path)          # ① 按扩展名识别类型
    doc_id = self._make_doc_id(file_path)          # ② SHA256前16位作为文档ID

    raw_texts: list[str] = []
    if doc_type == DocType.PDF:
        raw_texts = await self._parse_pdf(file_path)       # ③ PDF→文字+图片+表格
    elif doc_type == DocType.IMAGE:
        raw_texts = await self._parse_image(file_path)     # ④ 图片→OCR+视觉描述
    elif doc_type == DocType.TABLE:
        raw_texts = await self._parse_table(file_path)     # ⑤ 表格→结构化文本
    elif doc_type in (DocType.TEXT, DocType.MARKDOWN):
        raw_texts = self._parse_text(file_path)           # ⑥ 纯文本直接读取

    chunks = self._chunk_texts(raw_texts, doc_id, doc_type, file_path)  # ⑦ 分块
    return chunks

async def extract(self, chunks: list[DocumentChunk]) -> list[ExtractionResult]:
    results: list[ExtractionResult] = []
    for i in range(0, len(chunks), self.BATCH_SIZE):
        batch = chunks[i : i + self.BATCH_SIZE]  # 每5个一批
        for chunk in batch:
            result = await self._extract_from_chunk(chunk)  # 每个chunk独立抽取
            results.append(result)
    merged = self._deduplicate(results)  # 跨chunk去重
    return merged

@staticmethod
def _deduplicate(results: list[ExtractionResult]) -> list[ExtractionResult]:
    seen_entities: dict[str, Entity] = {}
    seen_relations: set[tuple[str, str, str]] = set()
    deduped: list[ExtractionResult] = []

    for result in results:
        # 实体去重：同名+同类型视为同一个实体
        unique_entities = []
        for ent in result.entities:
            key = f"{ent.name}::{ent.type}"
            if key not in seen_entities:
                seen_entities[key] = ent
                unique_entities.append(ent)

        # 关系去重：完全相同的三元组只保留一个
        unique_relations = []
        for rel in result.relations:
            key = (rel.head, rel.relation, rel.tail)
            if key not in seen_relations:
                seen_relations.add(key)
                unique_relations.append(rel)
        # ...
    return deduped
```

    └→ return 0  # 静默跳过，不影响图谱存储继续执行

异常场景3: Neo4j连接失败
  orchestrator/graph.py:132  except Exception: pass
    └→ entity_count保持0，不影响HTTP响应返回

异常场景4: LLM JSON解析失败
  agents/knowledge_extract_agent.py:141  except JSONDecodeError:
    └→ 返回空的 ExtractionResult(entities=[], relations=[], events=[])
         ↓
       后续流程正常继续（只是本轮没有抽取到知识）

  agents/doc_parser_agent.py:136  except Exception:
    └→ texts.append(f"[PDF 解析失败] {file_path}")
```

**容错设计原则**：每个环节的失败只影响自身，不阻断下游。入库流水线没有条件路由（始终线性END），即使部分环节失败，HTTP响应仍返回部分结果。

### 1. LangGraph入库流水线构建 (`orchestrator/graph.py:93-148`)

```python
def _build_ingest_graph(doc_parser, extractor, vector_store, knowledge_graph) -> StateGraph:

    async def parse_documents(state: dict) -> dict:
        file_paths = state.get("file_paths", [])
        chunks = await doc_parser.parse_batch(file_paths)  # ① 批量解析
        return {**state, "chunks": chunks}

    async def extract_knowledge(state: dict) -> dict:
        chunks = state.get("chunks", [])
        extractions = await extractor.extract(chunks)  # ② 知识抽取
        return {**state, "extractions": extractions}

    async def store_vectors(state: dict) -> dict:
        chunks = state.get("chunks", [])
        count = 0
        if vector_store and chunks:
            try:
                if vector_store.embeddings_available:
                    count = await vector_store.add_chunks(chunks)  # ③ 向量入库
            except Exception:
                pass
        return {**state, "vectors_stored": count}

    async def store_graph(state: dict) -> dict:
        extractions = state.get("extractions", [])
        entity_count = 0
        if knowledge_graph:
            try:
                for ext in extractions:
                    for ent in ext.entities:
                        await knowledge_graph.upsert_entity(ent)  # ④ 实体入库
                        entity_count += 1
                    for rel in ext.relations:
                        await knowledge_graph.add_relation(rel)  # ⑤ 关系入库
            except Exception:
                pass
        return {**state, "entities_stored": entity_count}

    # ⑥ 构建有向图: parse → extract → store_vectors → store_graph → END
    graph = StateGraph(dict)
    graph.add_node("parse", parse_documents)
    graph.add_node("extract", extract_knowledge)
    graph.add_node("store_vectors", store_vectors)
    graph.add_node("store_graph", store_graph)
    graph.set_entry_point("parse")
    graph.add_edge("parse", "extract")
    graph.add_edge("extract", "store_vectors")
    graph.add_edge("store_vectors", "store_graph")
    graph.add_edge("store_graph", END)
    return graph.compile()
```

### 3. 多模态解析+降级 (`agents/doc_parser_agent.py:121-156`)

```python
async def _parse_pdf(self, file_path: str) -> list[str]:
    """PDF解析：PyPDF2提取文字 → 失败则LLM视觉兜底"""
    texts: list[str] = []
    try:
        from PyPDF2 import PdfReader
        reader = PdfReader(file_path)
        for page in reader.pages:
            page_text = page.extract_text() or ""
            if page_text.strip():
                texts.append(page_text.strip())
    except Exception:
        texts.append(f"[PDF 解析失败] {file_path}")

    if not texts:  # ← 关键降级点：文字提取为空时自动切换视觉方案
        texts = await self._pdf_vision_fallback(file_path)
    return texts

async def _pdf_vision_fallback(self, file_path: str) -> list[str]:
    """LLM视觉兜底：将PDF页面渲染为图片 → LLM多模态理解"""
    from pdf2image import convert_from_path
    images = convert_from_path(file_path, dpi=150, first_page=1, last_page=5)
    texts = []
    for img in images:
        description = await self._describe_image_with_llm(img)
        texts.append(description)
    return texts
```

### 4. Chunking滑动窗口 (`agents/doc_parser_agent.py:262-286`)

```python
def _chunk_texts(self, texts, doc_id, doc_type, source) -> list[DocumentChunk]:
    chunks: list[DocumentChunk] = []
    idx = 0
    for text in texts:
        start = 0
        while start < len(text):
            end = start + self.CHUNK_SIZE          # ① 512字符窗口
            content = text[start:end]
            if content.strip():
                chunks.append(DocumentChunk(
                    content=content.strip(),
                    doc_id=doc_id,
                    chunk_index=idx,
                    doc_type=doc_type,
                    metadata={
                        "source": source,
                        "char_start": start,
                        "char_end": end
                    },
                ))
                idx += 1
            start = end - self.CHUNK_OVERLAP       # ② 64字符重叠滑动
    return chunks
```

async def upsert_entity(self, entity, version=1, source=""):
    """Neo4j MERGE：同一name的实体只建一个节点，后续更新而非新建"""
    cypher = """
    MERGE (e:Entity {name: $name})
    ON CREATE SET
        e.type = $type, e.description = $description,

            "name": entity.name, "type": entity.type,
            "description": entity.description, "version": version,
            "source": source, "now": int(time.time()),
        })

### 7. CDC差量计算 (`services/cdc_processor.py:100-124`)

```python
@staticmethod
def compute_diff(before: str, after: str) -> dict:
    """对比新旧文本，区分大改(>30%)和小改"""
    before_lines = before.splitlines() if before else []
    after_lines = after.splitlines() if after else []

    before_set = set(before_lines)
    after_set = set(after_lines)

    added = after_set - before_set
    removed = before_set - after_set

    change_ratio = len(added | removed) / max(len(before_lines) + len(after_lines), 1)

    return {
        "added_lines": list(added),
        "removed_lines": list(removed),
        "added_count": len(added),
        "removed_count": len(removed),
        "change_ratio": round(change_ratio, 4),
        "is_major_change": change_ratio > 0.3,  # 超过30%视为大改
    }
```

### 8. Embedding子进程隔离 (`services/embedding_worker.py:22-48`)

```python
def _worker_process(request_queue, response_queue):
    """独立进程中加载sentence-transformers，通过Queue通信"""
    from sentence_transformers import SentenceTransformer
    model = SentenceTransformer("shibing624/text2vec-base-chinese", device="cpu")
    model.encode("warmup")  # 预热

    while True:
        msg = request_queue.get(timeout=1)
        if msg is None or msg[0] == "shutdown":
            break
        msg_id, texts = msg
        vectors = model.encode(texts, show_progress_bar=False).tolist()
        response_queue.put((msg_id, vectors))
```
